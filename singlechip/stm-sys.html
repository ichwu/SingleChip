NVIC 结构体
	uint32_t   ISER[8]       //中断使能寄存器组
	uint32_t   ICER[8]       //中断除能寄存器组
	
	uint32_t   ISPR[8]       //中断挂起控制寄存器组
	uint32_t   ICPR[8]       //中断解挂控制寄存器组
	
	uint32_t   IABR[8]       //中断激活标志位寄存器组
	uint8_t     IP[240]        //中断优先级控制寄存器组
	uint32_t   STIP            //软件触发中断寄存器组   
	
	
ISER[0]中bit0~32  对应中断0~31
ISER[1]中bit0~32  对应中断32~63
ISER[2]中bit0~3    对应中断64~67
要使能那个中断  就置1相应位  随后还有配合中断分组，屏蔽，IO口映射等
ICER与ISER相反

ISPR 用来将相应位的中断挂起，从而执行同级或更高级的中断
ICPR与ISPR相反

IP-Interrupt Priority Registers  中断优先级寄存器
IP[0]~IP[67] 分别对应 中断0~67   使用8bit中的高4位
这高4位由SCB->AIRCR分为5个组  表示抢占优先级和响应优先级的位分配
		比如设置第3组 则抢占优先级占3位 可表示0~7
						响应优先级占1位 可表示0~1

void MY_NVIC_Init（u8 NVIC_PreemptionPriority，    //抢占优先级
					u8 NVIC_SubPriority，                 //响应优先级
					u8 NVIC_Channel，                     //中断编号  0~67 或 EXTI0_IRQn 等
					u8 NVIC_Group                            //中断分组：0~4
					）
					

EXTI外部中断结构体
	uint32_t IMR          //中断屏蔽寄存器     前19位有效 置1时则开启相应线上的中断 置0则关闭
	uint32_t EMR         //事件屏蔽寄存器     同IMR  只针对事件
	uint32_t RTSR        //上升沿触发选择寄存器  
	uint32_t FTSR        //下降沿触发选择寄存器
	uint32_t SWIER      //软件中断事件寄存器  置1时 若设置了IMR和EMR 则产生一次中断 否则挂起PR相应位 
	uint32_t PR            //挂起寄存器    类似中断标志 
	
因为外部中断线只有16个 而IO口有16*7个    每条中断线对应7个IO口  

EXTICR外部中断配置寄存器结构组
	uint32_t EVCR
	uint32_t MAPR
	uint32_t EXTICR[4]
EXTICR[0]~[3]   只用低16位 每4bit决定使用A B C D E F G那个IO口 

void Ex_NVIC_Config（u8 GPIOx，u8 BITx，u8 TRIM）     //TRIM=01 下降沿    02  上升沿   03  任意
